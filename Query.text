CREATE TABLE RETAIL_TB (
    RECORDID      VARCHAR2(12) PRIMARY KEY,
    CUSTOMERNAME  VARCHAR2(40) NOT NULL,
    PRODUCTNAME   VARCHAR2(40) NOT NULL,
    PURCHASE_DATE DATE NOT NULL,
    QUANTITY      NUMBER,
    PRICE         NUMBER,
    REMARKS       VARCHAR2(100)
);


CREATE SEQUENCE RETAIL_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99;

DESC RETAIL_TB;

SELECT * FROM RETAIL_TB;
